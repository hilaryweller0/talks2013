\heading{CLUST: \normalsize\normalfont Continuous Linear-Upwind\\ Stabilised Transport (Weller, 2012)}

$q_e = b
\underbrace{\biggl(
    q_u + (\mathbf{x}_e-\mathbf{x}_u)\cdot \nabla_u q
\biggr)}_{\text{linear upwind}}
+ (1-b)
\underbrace{
    \half (q_u + q_d)
}_{\text{centred}}
$

where $b = \frac{1}{2}\frac{|u_n|}{|\mathbf{u}_e|}$
